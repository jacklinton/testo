/**
 * @module uuid
 */
/**
 * This is a middleware module that takes in the request and response, adds a uuid property and
 * value to the request, and then adds a header property to the response with the said UUID.
 */
import uuidv4 from 'uuid/v4.js'

const requestIdHeader = 'X-Request-Id'

const uuid = (req, res, next) => {
  req.uuid = uuidv4()
  res.set(requestIdHeader, req.uuid)
  next()
}

export default uuid
